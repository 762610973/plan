- [merge-two-sorted-lists](https://leetcode.cn/problems/merge-two-sorted-lists/)
```go
package main
type ListNode struct {
	Val int
	Next *ListNode
}
func mergeTwoLists(list1 *ListNode, list2 *ListNode) *ListNode {
	res := &ListNode{}
	cur := res
	for list1 != nil && list2 != nil {
	    if list1.Val > list2.Val {
		    cur.Next = &ListNode{Val: list2.Val}
		    list2 = list2.Next
		} else {
			cur.Next = &ListNode{Val: list1.Val}
			list1 = list1.Next
		}
		cur = cur.Next
	}
	if list1 == nil {
		cur.Next = list2
	}
	if list2 == nil {
		cur.Next = list1
	}
	return res.Next
}
// 递归解法
func mergeTwoListsRecursion(list1 *ListNode, list2 *ListNode) *ListNode {
	if  list1 == nil {
		return list2
	}
	if list2 == nil {
		return list1
	}
	if list1.Val < list2.Val {
		list1.Next = mergeTwoListsRecursion(list1.Next,list2)
		return list1
	} else {
		list2.Next = mergeTwoListsRecursion(list1,list2.Next)
		return list2
	}
}
```

- [two-sum](https://leetcode.cn/problems/two-sum/)

```go
package main
func twoSum(nums []int, target int) []int {
    // key: 数组中的元素,val: 元素的下标
    m := make(map[int]int,len(nums))
    for index,val := range nums {
        t := target-val
        if idx,b := m[t];b {
            return []int{idx,index}
        }
        m[val] = index
    }

    return nil
}
```

- [add-two-numbers](https://leetcode.cn/problems/add-two-numbers/)
```go
package main
type ListNode struct {
	Val int
	Next *ListNode
}
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    res := &ListNode{}
    cur := res
    var extra int
    var v1,v2 int
    for l1 != nil || l2 != nil || extra == 1{
        if l1 == nil {
            v1 = 0
        } else {
            v1 = l1.Val
            l1 = l1.Next
        }
        if l2 == nil {
            v2 = 0
        } else {
            v2 = l2.Val
            l2 = l2.Next
        }
        now := v1 + v2 + extra
        if now >= 10 {
            now = now % 10
            extra = 1
        } else {
            extra = 0
        }
        cur.Next = &ListNode {
            Val: now,
        }
        cur = cur.Next
    }

    return res.Next
}
```
